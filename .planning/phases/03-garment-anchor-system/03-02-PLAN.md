<objective>
Build the BodyMeasurement computation layer, AnchorResolver dispatch, and all four anchor strategies (shirt, abaya, pants, accessory). This is the core algorithmic heart of Phase 3.

Purpose: Each garment type needs purpose-built positioning logic. BodyMeasurement extracts body proportions from landmarks once per frame. Each strategy consumes those measurements to produce position/scale/rotation for the 3D model. The AnchorResolver dispatches to the correct strategy by garment type.
Output: 6 production files + 6 test files covering the full anchor computation pipeline.
</objective>

```typescript
export interface AnchorStrategy {
  compute(
    measurements: BodyMeasurements,
    config: GarmentConfig,
    modelDims: { w: number; h: number; d: number },
  ): AnchorResult | null;
}

export interface AnchorResult {
  position: { x: number; y: number; z: number };
  scale: { x: number; y: number; z: number };
  rotationY: number;
  confidence: number;
  degraded: boolean;
}

export interface BodyMeasurements {
  shoulderWidthMetric: number;
  torsoHeightMetric: number;
  hipWidthMetric: number;
  shoulderToAnkleMetric: number;
  hipToAnkleMetric: number;
  shoulderCenter: { x: number; y: number };
  hipCenter: { x: number; y: number };
  kneeCenter: { x: number; y: number } | null;
  ankleCenter: { x: number; y: number } | null;
  nosePosition: { x: number; y: number } | null;
  earMidpoint: { x: number; y: number } | null;
  visibility: Record<number, number>;
  bodyTurnY: number;
  shoulderTiltZ: number;
  hasShoulders: boolean;
  hasHips: boolean;
  hasKnees: boolean;
  hasAnkles: boolean;
  hasHead: boolean;
}

export interface GarmentConfig {
  type: GarmentType;
  requiredLandmarks: number[];
  optionalLandmarks: number[];
  visibilityThreshold: number;
  widthPadding: number;
  heightPadding: number;
  verticalOffset: number;
  widthRef: 'shoulder' | 'hip' | 'ear';
  heightRef: 'torso' | 'shoulder_to_ankle' | 'hip_to_ankle' | 'head';
}
```
